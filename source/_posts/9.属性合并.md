## 将具有相同属性的对象合并到一个数组中

<!-- more -->



新年新气象，新的一年，继续加油。

```javascript
    let data = [
        {
            'group_name':'A',
            'type': 'apple'
        },
        {
            'group_name':'A',
            'type': 'almond'
        },
        {
            'group_name':'B',
            'type': 'banana'
        },
        {
            'group_name':'B',
            'type': 'bennet'
        }
    ];

    //代码实现
    // { group_name, type } = currValue
    const res = data.reduce( (prv, { group_name, type } ) => {
        const exist = prv.find(item => item.group_name == group_name);
        if (exist) {
            exist["data"].push({
                group_name:group_name,
                type:type
            });
        } else {
            prv.push({ group_name, data: [{
                group_name:group_name,
                type:type
            }]})
        }
        return prv;
    }, []);

    console.log(res)

    // 打印结果如下：
    [
        {  
           'group_name':A,
            data:[
                {
                    'group_name':'A',
                    'type': 'apple'
                },
                {
                    'group_name':'A',
                    'type': 'almond'
                },
            ]
        },
        {  
            'group_name':B,
             data:[
                {
                    'group_name':'B',
                    'type': 'banana'
                },
                {
                    'group_name':'B',
                    'type': 'bennet'
                }
            ]
        }
    ]
```
